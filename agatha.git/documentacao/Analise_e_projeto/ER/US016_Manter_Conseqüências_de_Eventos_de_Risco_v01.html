Table tb_consequencia

Columns
Tabela responsável por armazenar os dados de consequência de eventos de riscos.
* id_consequencia serial Identificador da consequência de evento de risco
* ds_consequencia varchar( 200 ) Descrição da consequência de eventos de riscos
* ic_ativo bool DEFAULT true Situação da consequência
Indexes
pk_tb_consequencia ON id_consequencia
